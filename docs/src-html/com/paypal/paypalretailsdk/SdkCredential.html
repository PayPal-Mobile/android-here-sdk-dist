001package com.paypal.paypalretailsdk;
002
003import android.util.Log;
004
005public class SdkCredential {
006  public static final String liveService = "live";
007  public static final String sandboxService = "sandbox";
008  public static final String stageService = "stage";
009  private static final String LOG_TAG = SdkCredential.class.getSimpleName();
010
011  String accessToken;
012  String refreshUrl;
013  String refreshToken;
014  String clientId;
015  String clientSecret;
016  String environment;
017  String repository;
018
019  /**
020   * Build SDK Credentials
021   * @param environment Executing environment for the SDK e.g. live, sandbox, stage2d0065, etc.
022   * @param accessToken Access token for authentication
023   */
024  public SdkCredential(String environment, String accessToken, String softwareRepository) {
025    this.environment = environment;
026    this.accessToken = accessToken;
027    this.repository = softwareRepository;
028  }
029
030  /**
031   * Build SDK Credentials
032   * @param environment Executing environment for the SDK e.g. live, sandbox, stage2d0065, etc.
033   * @param accessToken Access token for authentication
034   */
035  public SdkCredential(String environment, String accessToken) {
036    this.environment = environment;
037    this.accessToken = accessToken;
038    if (environment.contains(liveService))
039    {
040      this.repository = "production";
041    }
042    else if (environment.contains(sandboxService))
043    {
044      this.repository = "sandbox";
045    }
046    else if (environment.contains(stageService))
047    {
048      this.repository = "dev-stage-1";
049    }
050    else
051    {
052      Log.e(LOG_TAG, "No repository defined for this environment: " + environment);
053      throw new RuntimeException("Proper environment required!");
054    }
055  }
056
057  /**
058   * URL to refresh an expired access token. The JSON response must contain the access token set to
059   * access_token property
060   */
061  public SdkCredential setTokenRefreshCredentials(String refreshUrl) {
062    this.refreshUrl = refreshUrl;
063    return this;
064  }
065
066  /**
067   * Provide credentials to refresh an expired access token using refresh token and client Id & secret
068   */
069  public SdkCredential setTokenRefreshCredentials(String refreshToken, String clientId, String clientSecret) {
070    this.refreshToken = refreshToken;
071    this.clientId = clientId;
072    this.clientSecret = clientSecret;
073    return this;
074  }
075}
